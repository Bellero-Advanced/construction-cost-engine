      xlabel={เดือน (เดือนที่ 1 = ส.ค.\ 2568)},
      ylabel={CMI (จุด, ฐาน 2563 $=$ 100)},
      xmin=0.5, xmax=9.5,
      ymin=108, ymax=115,

  \label{fig:chart-tpso}
