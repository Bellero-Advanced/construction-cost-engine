\begin{figure}[H]
  \centering
  \begin{tikzpicture}[
    font=\footnotesize,
    node distance=0.7cm and 1.2cm,
    box/.style={draw,rounded corners=3pt,minimum height=0.9cm,minimum width=3.2cm,align=center,fill=blue!8},
    tool/.style={draw,rounded corners=3pt,minimum height=0.7cm,minimum width=2.6cm,align=center,fill=orange!12,font=\scriptsize},
    decision/.style={diamond,draw,aspect=2.5,minimum width=3.0cm,align=center,fill=yellow!15,font=\scriptsize},
    arr/.style={-{Stealth[length=2mm]},thick},
    darr/.style={-{Stealth[length=2mm]},thick,dashed,gray!60},
  ]
    % Main loop nodes
    \node[box,fill=green!15] (task) {รับ Task\\``หาราคา [วัสดุ X] จาก [แหล่ง Y]''};
    \node[box,below=of task] (plan) {LLM วางแผน\\(เลือกคำค้น, กลยุทธ์)};
    \node[box,below=of plan] (act) {เรียก Tool};
    \node[decision,below=0.8cm of act] (valid) {ราคาสมเหตุ\\สมผล?};
    \node[box,below=0.9cm of valid,fill=green!20] (submit) {POST /api/admin/ai-agent\\(บันทึก live + history)};
    \node[box,right=2.5cm of valid,fill=red!10] (retry) {ปรับคำค้น\\หรือแหล่งอื่น};

    % Tool nodes (right side)
    \node[tool,right=2.5cm of act] (t1) {web\_search(query)};
    \node[tool,below=0.3cm of t1] (t2) {browse\_url(url)};
    \node[tool,below=0.3cm of t2] (t3) {extract\_price(html)};

    % Arrows
    \draw[arr] (task) -- (plan);
    \draw[arr] (plan) -- (act);
    \draw[arr] (act) -- (valid);
    \draw[arr] (valid) -- node[right,xshift=2pt]{ผ่าน} (submit);
    \draw[arr] (valid) -- node[above,font=\scriptsize]{ไม่ผ่าน} (retry);
    \draw[arr] (retry) |- (plan);

    % Tool connections
    \draw[darr] (act) -- (t1);
    \draw[darr] (act) -- (t2);
    \draw[darr] (act) -- (t3);
    \draw[darr] (t3.west) -- +(-0.4,0) |- (valid.east);
  \end{tikzpicture}
  \caption{วงจรการทำงานของ AI Agent Research (Observe-Plan-Act-Validate loop). LLM ทำหน้าที่เป็น reasoning engine ที่ควบคุมการเรียก tool และตรวจสอบความสมเหตุสมผลของราคาก่อนบันทึกเข้าระบบ.}
  \label{fig:agent-loop}
\end{figure}
